# Способности

```dataview
TABLE WITHOUT ID
  file.link as Способность,
  тип as Тип,
  стоимость_ос as "ОС",
  мана as "MP"
FROM "Способности"
WHERE contains(this.способности_выбранные, file.name)
SORT уровень ASC
```
