# Снаряжение

```dataview
TABLE WITHOUT ID
  file.link as Оружие,
  урон as Урон,
  свойства as Свойства
FROM "Снаряжение"
WHERE тип = "Оружие" AND contains(this.снаряжение_выбранное, file.name)
```

```dataview
TABLE WITHOUT ID
  file.link as Броня,
  бонус_кб as "Бонус КБ",
  свойства as Свойства
FROM "Снаряжение"
WHERE тип = "Броня" AND contains(this.снаряжение_выбранное, file.name)
```

# Способности

```dataview
TABLE WITHOUT ID
  file.link as Способность,
  тип as Тип,
  стоимость_ос as "ОС",
  мана as "MP"
FROM "Способности"
WHERE contains(this.способности_выбранные, file.name)
SORT уровень ASC
```
